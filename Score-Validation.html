ci95 <- function(x) {
  x <- x[!is.na(x)]
  n <- length(x)
  if (n <= 1) return(c(M = mean(x), LCL = NA_real_, UCL = NA_real_))
  m <- mean(x)
  se <- sd(x) / sqrt(n)
  tcrit <- qt(0.975, df = n - 1)
  c(M = m, LCL = m - tcrit * se, UCL = m + tcrit * se)
}

ci_mean_t <- function(x, level = 0.95) {
  x <- x[is.finite(x)]
  n <- length(x)
  m <- mean(x)
  if (n <= 1) return(c(M = m, LCL = NA_real_, UCL = NA_real_))
  se <- sd(x) / sqrt(n)
  tcrit <- qt(1 - (1 - level) / 2, df = n - 1)
  c(M = m, LCL = m - tcrit * se, UCL = m + tcrit * se)
}


PrRange <- map_dfr(unique_scores, function(s) {
  filtered <- old_score_df %>%
    filter(Outcome == 1, !is.na(score), score >= s)

  if (nrow(filtered) == 0) {
    return(tibble(score = s, M = NA_real_, LCL = NA_real_, UCL = NA_real_))
  }

  first_trigger <- filtered %>%
    group_by(id) %>%
    summarise(DaysBfOutcome = max(DaysBfOutcome, na.rm = TRUE), .groups = "drop") %>%
    filter(is.finite(DaysBfOutcome)) %>%
    pull(DaysBfOutcome)

  if (length(first_trigger) == 0) {
    return(tibble(score = s, M = NA_real_, LCL = NA_real_, UCL = NA_real_))
  }

  stats <- ci_mean_t(first_trigger)
  tibble(score = s, M = stats["M"], LCL = stats["LCL"], UCL = stats["UCL"])
})

work <- freq_table %>%
  arrange(desc(score)) %>%
  mutate(
    TP = cumsum(deaths),
    FP = cumsum(survived)
  )

# Total
tot_dead <- sum(work$deaths, na.rm = TRUE)
tot_surv <- sum(work$survived, na.rm = TRUE)

# TN and FN
work <- work %>%
  mutate(
    FN = tot_dead - TP,
    TN = tot_surv - FP
  )

# 95%CI (asymptotic)
se_ci <- binom.confint(work$TP, 
                       work$TP + work$FN, 
                       methods = "asymptotic")
sp_ci <- binom.confint(work$TN, 
                       work$TN + work$FP, 
                       methods = "asymptotic")

Sens_Sp <- work %>%
  transmute(
    score,
    TP, FP, TN, FN,
    Sensitivity = 100 * TP / (TP + FN),
    Specificity = 100 * TN / (TN + FP),
    LCL_Se = 100 * se_ci$lower,
    UCL_Se = 100 * se_ci$upper,
    LCL_Sp = 100 * sp_ci$lower,
    UCL_Sp = 100 * sp_ci$upper
  ) %>%
  arrange(score)
